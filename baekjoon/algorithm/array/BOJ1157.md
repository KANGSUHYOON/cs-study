#include <iostream>
#include <string>
using namespace std;

int main() {
    string s;
    cin >> s;

    int cnt[26] = {0};

    for (int i = 0; i < s.length(); i++) {
        char c = s[i];

        if (c >= 'a' && c <= 'z') {
            c = c - 'a' + 'A';
        }

        cnt[c - 'A']++;
    }

    int maxCnt = 0;
    char result = '?';

    for (int i = 0; i < 26; i++) {
        if (cnt[i] > maxCnt) {
            maxCnt = cnt[i];
            result = 'A' + i;
        }
        else if (cnt[i] == maxCnt) {
            result = '?';
        }
    }

    cout << result;
    return 0;
}

    int cnt[26] = { 0 };

    for (int i = 0; i < word.length(); i++) {

        cnt[c - 'A']++;
    }

    int maxVal = 0;
    int idx = -1;
    bool duplicate = false;

    for (int i = 0; i < 26; i++) {
        if (cnt[i] > maxVal) {
            maxVal = cnt[i];
            idx = i;
            duplicate = false;
        }
        else if (cnt[i] == maxVal && cnt[i] != 0) {
            duplicate = true;
        }
    }

    if (duplicate)
        cout << "?";
    else
        cout << char(idx + 'A');

    return 0;
}
